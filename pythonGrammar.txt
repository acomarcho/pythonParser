ASSIGNMENT -> _IDENTIFIER = DATA | _IDENTIFIER = _IDENTIFIER
DATA -> STRING | _INTEGER | FLOAT | LIST | TUPLE | SET | DICTIONARY | BOOLEAN | _IDENTIFIER
LIST -> [ ] | [ LIST_DATA ]
LIST_DATA -> DATA | DATA , LIST_DATA
FLOAT -> _INTEGER . | _INTEGER . _INTEGER
STRING -> " _ANY " | ' _ANY '
TUPLE -> ( ) | ( DATA , ) | ( TUPLE_DATA )
TUPLE_DATA -> DATA | DATA , TUPLE_DATA
SET -> { SET_DATA }
SET_DATA -> DATA | DATA , SET_DATA
DICTIONARY -> { } | { DICT_DATA }
DICT_DATA -> STRING : DATA , | STRING : DATA
BOOLEAN -> True | False
